  \item[\textcolor{blue}{\ding{43}}]\step{\textcolor{black}{\textbf{What does it mean $\log x$?}}}\step{\quad It is the \emph{natural logarithm} of $x$};
\bigskip  \item[\textcolor{blue}{\ding{43}}]\step{Racall that the logarithm in base $a$ of $b$ is that number $t$ s.t. $a^{t}=b$};\\
 \item[\textcolor{blue}{\ding{43}}]\step{Therefore $t=\log_ab$ means that $a^t=b$};
\step{\hspace*{2cm} for example $\log_28=3$ since $2^3=8$}\\
\step{\hspace*{2cm} and $\log_5625=4$ since $5^4=625$}.

\item[\textcolor{blue}{\ding{43}}]\step{When the base $a=e=2,7182818284590\cdots$ is the Nepier number,}\\
\step{the logarithm in base $e$ is called \emph{natural logarithm}};

\item[\textcolor{blue}{\ding{43}}]\step{\textcolor{green}{hence}
$\log10=2.30258509299404568401$ since $e^{2.30258509299404568401}=10$}
\item[\textcolor{blue}{\ding{43}}]\step{finally $\log x$ is a function}
\end{itemize}}

\heading{GREAT OPEN PROBLEM AT THE END OF:}\bigskip\pause

\parstepwise{\begin{itemize}
  \item[\textcolor{blue}{\ding{43}}]\step{To prove the \emph{Conjecture} of Legendre -- Gau\ss\ \ }
\restep{{{$\displaystyle{\pi(x)\sim\frac{x}{\log x}}$ if $x\rightarrow\infty$}}}
  \item[\textcolor{blue}{\ding{43}}]\step{that is:\ \ }
\restep{{{$\displaystyle{\left|\frac{\pi(x)}{\frac{x}{\log x}}-1\right|\rightarrow0}$ if $x\rightarrow\infty$}}}\bigskip
  \item[\textcolor{blue}{\ding{43}}]\step{that is:\ \ }
\restep{{{$\displaystyle{\left|\pi(x)-\frac{x}{\log x}\right|}$ is ``much smaller'' than
$\displaystyle{\frac{x}{\log x}}$ if  $x\rightarrow\infty$}}}
 \item[\textcolor{blue}{\ding{43}}]\step{that is:\ \ }
\restep{$\displaystyle{\left|\pi(x)-\frac{x}{\log x}\right|=o\left(\frac{x}{\log x}\right)}$ if  $x\rightarrow\infty$}
  \item[\textcolor{blue}{\ding{43}}]\step{that is (to say it \`a la  Gau\ss):}
\restep{{{$\displaystyle{|\pi(x)-\operatorname{li}(x)|}=o\left(\operatorname{li}(x)\right)$ if  $x\rightarrow\infty$}}}
\end{itemize}}
