\centerline{\bf PROBLEMI D'ESAME DI TEORIA DI GALOIS}
\bigskip
\bigskip\bigskip\bigskip

\item{A.} Sia $G={\rm GL}_2({\bf F}_5)$ il gruppo delle matrici invertibile due per due
a coefficienti in ${\bf F}_5$. \bigskip

\itemitem{1.} Dimostrare che l'ordine di $G$ \`e $480$;
\itemitem{2.} Dimostrare che $\pmatrix{1 & 1 \cr 0 & 1}$ genera un sottogruppo di
Sylow di $G$ e dire se si tratta di un sottogruppo normale;
\itemitem{3.}  Calcolare un $p$-sottogruppo di Sylow di ${\rm GL}_2({\bf F}_p)$
per ogni fissato numero primo $p$;
\itemitem{4.} Calcolare l'ordine dell' elemento $\pmatrix{2 & 0 \cr 0 & 1}$ in $G$;
\itemitem{5.} Sia $B$ l'insieme delle matrici diagonali di $G$. Dimostrare che $B$
\`e un sottogruppo di $G$ e calcolarne l'ordine;
\itemitem{6.} Dimostrare che
$$C=B\cup\left\{\pmatrix{0 & a \cr b & 0}\ | a,b\in {\bf F}_5^*\right\}$$
\`e un 2-Sylow di $G$ e dire se \`e un sottogruppo normale.
\itemitem{7.} Determinare un $3$-sottogruppo di Sylow di $G$ e dire se \`e normale.\break
\it Suggerimento: Dimostrare che ${\rm SL}_2({\bf F}_5)$ contiene un $3$-sottogruppo di
Sylow e osservare che se $a^3=\pmatrix{1 & 0 \cr 0 & 1}$ allora $a^2=a^{-1}$.
\bigskip\bigskip

\rm \item{B.} Sia $f(t)=t^6-8t^3+15$.\bigskip

\itemitem{1.} Determinare il campo di spezzamento ${\bf Q}(f)$ di $f$;
\itemitem{2.} Determinare tutti gli elementi di ${\rm Gal}({\bf Q}(f)/{\bf Q})$;
\itemitem{3.} Dimostrare che ${\bf Q}(f)$ \`e un estensione di Galois di ${\bf Q}(\sqrt{-3})$ e
dimostrare che ${\rm Gal}({\bf Q}(f)/{\bf Q}(\sqrt{-3}))$ \`e isomorfo a ${\bf Z}/3{\bf Z}\ \times
\ {\bf Z}/3{\bf Z}$;
\itemitem{4.} Determinare tutti i sottocampi di ${\bf Q}(f)$ che sono estensioni cubiche di
${\bf Q}$;
\itemitem{5.} Trovare un elemento $\alpha\in{\bf Q}(f)$ tale che ${\bf Q}(f)=
{\bf Q}(\alpha)$.
